# Synchro-Slider

## Usage

```js
$('.synchro_slider').SynchroSlider();
```

```js
$('.synchro_slider').SynchroSlider({
  'baseWidth' : '960',
  'baseHeight': '522',
  'minWidth'  : '860',
  'minHeight' : '340',
  'thumbWidth': '192',
  'thumbView' : '5',
  'thumbpath' : './images/',
  'slideSpeed': '500',
  'delayTime' : '4000',
  'easing'    : 'easeInOutQuad',
  'autoPlay'  : '1',
  'flickMove' : '1',
  'btnOpacity': '1',
  'pnOpacity' : '1'
});
```
